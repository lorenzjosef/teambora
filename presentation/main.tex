\begin{frame}{The assignment}
  \begin{columns}[T,totalwidth=\textwidth]
    \begin{column}{0.55\textwidth}
      \Large Design an AI-powered solution that helps municipalities identify, reach, and meaningfully connect lonely residents.
      \vspace{0.45cm}

      \callout{Without diagnosis. Without surveillance. Without replacing human connection.}
    \end{column}
    \begin{column}{0.38\textwidth}
      \begin{block}{Judges care about}
        \begin{itemize}
          \item Consent and dignity
          \item Real human connection
          \item GDPR + AI Act survival
          \item EUR 2M / 18-month feasibility
        \end{itemize}
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}[shrink=8]{The pain: outreach misses the people who avoid the label}
  \begin{columns}[T,totalwidth=\textwidth]
    \begin{column}{0.48\textwidth}
      \screenshotbox{5.1cm}{4.6cm}

    \end{column}
    \begin{column}{0.48\textwidth}
      \Large People may not say: ``I am lonely.''

      \vspace{0.28cm}
      But they will say:
      \begin{itemize}
        \item ``I am free Thursday evening.''
        \item ``I would join a small walk.''
        \item ``Only public places, please.''
      \end{itemize}

      \vspace{0.25cm}
      \muted{Municipal programs exist, but the uptake problem is a coordination problem.}
    \end{column}
  \end{columns}

\begin{frame}{Our reframing}
  \vspace{0.6cm}
  \begin{center}
    {\fontsize{26}{32}\selectfont Do not identify lonely people.}

    \vspace{0.45cm}
    {\fontsize{30}{36}\selectfont \alert{Identify opt-in demand for connection.}}
  \end{center}
  \vfill
  \begin{block}{Why this matters}
    The municipality does not need an individual loneliness score. It needs to know where safe, public, low-pressure connection can be made easier.
  \end{block}
\end{frame}

\begin{frame}{Privacy and dignity by design}
  \small
  \textbf{Resident controls:} alias-first, comfort settings, public-place defaults, report/block/cancel.

  \vspace{0.08cm}
  \textbf{Hard no's:} no loneliness score, no mental-health inference, no exact address exposure, no individual municipal view.
